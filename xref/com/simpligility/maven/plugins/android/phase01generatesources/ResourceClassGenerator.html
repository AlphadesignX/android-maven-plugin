17  /**
18   * Generates R classes containing appropriate resource values for dependent libraries.
19   *
20   * @author William Ferguson - william.ferguson@xandar.com.au
21   */
22  final class ResourceClassGenerator
23  {
24      private final GenerateSourcesMojo mojo;
25      private final File targetDirectory;
26      private final File genDirectory;
27      private final Log log;
28      private final ILogger androidUtilsLog;
29  
30      ResourceClassGenerator( final GenerateSourcesMojo mojo, final File targetDirectory,
31                              final File genDirectory )
32      {
33          this.mojo = mojo;
34          this.targetDirectory = targetDirectory;
35          this.genDirectory = genDirectory;
36          this.log = mojo.getLog();
37          this.androidUtilsLog = new MavenILogger( log );
38      }
39  
40      /**
41       * see {@link com.android.builder.core.AndroidBuilder#processResources(com.android.builder.internal.aapt.Aapt,
42       * com.android.builder.internal.aapt.AaptPackageConfig.Builder, boolean)}
43       *
44       * @param libraries
45       * @throws MojoExecutionException
46       */
47      public void generateLibraryRs( final Set<Artifact> libraries ) throws MojoExecutionException
48      {
49          // list of all the symbol tables
50          final List<SymbolTable> symbolTables = new ArrayList<>( libraries.size() );
51  
52          // For each dependency, load its symbol file.
53          for ( final Artifact lib : libraries )
54          {
55              final File unpackedLibDirectory = mojo.getUnpackedLibFolder( lib );
56              final File rFile = new File( unpackedLibDirectory, "R.txt" );
57  
58              if ( rFile.isFile() )
59              {
60                  final File libManifestFile = new File( unpackedLibDirectory, "AndroidManifest.xml" );
61                  final String packageName = new DefaultManifestParser( libManifestFile ).getPackage();
62                  log.info( "Reading R for " + packageName + " at " + rFile );
63  
64                  SymbolTable libSymbols = SymbolIo.read( rFile );
65                  libSymbols = libSymbols.rename( packageName, libSymbols.getTableName() );
66                  symbolTables.add( libSymbols );
67              }
68          }
69  
70          if ( symbolTables.isEmpty() )
71          {
72              return;
73          }
74  
75          // load the full resources values from the R.txt calculated for the project.
76          final File projectR = new File( targetDirectory, "R.txt" );
77          final SymbolTable mainSymbols = SymbolIo.read( projectR );
78  
79          // now loop on all the package name, merge all the symbols to write, and write them
80          RGeneration.generateRForLibraries( mainSymbols, symbolTables, genDirectory.getAbsoluteFile(), false );
81      }
